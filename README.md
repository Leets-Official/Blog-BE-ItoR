    │   │           │   ├── config
    │   │           │   ├── exception
    │   │           │   ├── response
    │   │           │   │   └── page

    │   │           ├── security
    │   │           │   └── jwt
    │   │           └── workspace
    │   │               ├── adapter

    │   │               │   │   └── web

    │   │               │   └── out
